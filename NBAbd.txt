USE master;

DROP DATABASE NBA;

CREATE DATABASE NBA;

USE NBA;

CREATE TABLE Teams (
    Id INT IDENTITY(1, 1) PRIMARY KEY,
    Name VARCHAR(100) NOT NULL,
    City VARCHAR(300) NOT NULL
);

CREATE TABLE Players (
    Id INT IDENTITY(1, 1) PRIMARY KEY,
    TeamId INT FOREIGN KEY REFERENCES Teams(Id),
    FullName VARCHAR(200) NOT NULL,
    Number INT
);
